int main() {
    ifstream file("demo.txt");
    if (!file.is_open()) {
        cerr << "Error opening the file!" << endl;
        return 1;
    }

    string line;
    while (getline(file, line)) {
        if (isKeyword(line))
            cout << line << " is a KEYWORD" << endl;
        else if (isValid(line))
            cout << line << " is VALID" << endl;
        else
            cout << line << " is INVALID" << endl;
    }

    file.close();
    return 0;
}
